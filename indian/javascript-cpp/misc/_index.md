```js
/* Web Worker*/
const AsposePageWebWorker = new Worker("AsposePageforJS.js");
```

```html
<!-- Load and initiate Aspose.Page for JavaScript via C++ -->
<script type="text/javascript" async src="AsposePageforJS.js"></script>
```
